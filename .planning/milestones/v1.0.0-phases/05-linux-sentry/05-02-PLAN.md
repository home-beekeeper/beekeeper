func isEditorDescendant(pid uint32, tree map[uint32]ProcessNode) bool {
    editorNames := map[string]bool{
        "code": true, "code-insiders": true,
        "cursor": true, "windsurf": true, "codium": true,
    }
    for pid != 0 {
        node, ok := tree[pid]
        if !ok { break }
        if editorNames[filepath.Base(node.Exe)] { return true }
        pid = node.PPID
    }
    return false
}

    type EventKind uint8
    const (
        EventProcessCreate EventKind = iota
        EventFileAccess
        EventNetworkConnect
    )

    type SentryEvent struct {
        Kind      EventKind  `json:"kind"`
        PID       uint32     `json:"pid"`
        PPID      uint32     `json:"ppid"`
        UID       uint32     `json:"uid"`
        Exe       string     `json:"exe"`
        Cmdline   string     `json:"cmdline,omitempty"`
        FilePath  string     `json:"file_path,omitempty"`   // EventFileAccess only
        DstAddr   net.IP     `json:"dst_addr,omitempty"`    // EventNetworkConnect only
        DstPort   uint16     `json:"dst_port,omitempty"`    // EventNetworkConnect only
        KTimeNS   uint64     `json:"ktime_ns"`
        WallTime  time.Time  `json:"wall_time"`
    }

    type ProcessNode struct {
        PID      uint32
        PPID     uint32
        UID      uint32
        Exe      string
        Cmdline  string
        SeenAt   time.Time
    }

    type SentryAlert struct {
        RuleID              string
        RuleName            string
        Severity            string    // "critical" or "high"
        BaselineMode        bool
        ProcessPID          uint32
        ProcessExe          string
        ParentChain         []string
        FilesAccessed       []string
        NetworkDests        []string
        CorrelatedExtension string
        QuarantineRec       bool
        Timestamp           time.Time
    }

    // RuleWindowEntry tracks events within a time window for a single rule.
    type RuleWindowEntry struct {
        PID      uint32
        Value    string    // file path, domain, or CLI name depending on rule
        SeenAt   time.Time
    }

    // RuleState holds per-rule sliding-window state. Owned by the daemon goroutine;
    // never shared across goroutines. The correlation engine receives a pointer and
    // updates it in place.
    type RuleState struct {
        // SENTRY-001: file access events per PID within 60s
        CredAccessByPID map[uint32][]RuleWindowEntry
        // SENTRY-002: credential CLI spawns per PID within 60s
        CredCLIByPID map[uint32][]RuleWindowEntry
        // SENTRY-003: phone-home connections per PID within 10 min
        PhoneHomeByPID map[uint32][]RuleWindowEntry
        // SENTRY-004/005: recent alert timestamps for correlation
        RecentAlerts []recentAlert
    }

    type recentAlert struct {
        RuleID    string
        PID       uint32
        FiredAt   time.Time
    }

    // RuleConfig holds tunable thresholds. Zero-value uses the defaults.
    type RuleConfig struct {
        CredAccessThreshold  int           // default 2
        CredCLIThreshold     int           // default 2
        PhoneHomeWindowMin   time.Duration // default 10 min
        CredAccessWindowSec  time.Duration // default 60s
        CredCLIWindowSec     time.Duration // default 60s
        FreshExtWindowMin    time.Duration // default 30 min
        ExfilFusionWindowMin time.Duration // default 5 min
    }

    // InventorySnapshot is a point-in-time view of recently installed extensions,
    // provided to the correlation engine by the daemon goroutine.
    type InventorySnapshot struct {
        // RecentExtensions maps extension ID to install time.
        RecentExtensions map[string]time.Time
    }

    // NewRuleState initializes a zero RuleState.
    func NewRuleState() *RuleState {
        return &RuleState{

            CredCLIByPID:    make(map[uint32][]RuleWindowEntry),
            PhoneHomeByPID:  make(map[uint32][]RuleWindowEntry),
        }
    }

    package sentry

    import (
        "encoding/json"
        "errors"
        "os"
        "time"
    )

    type BaselineState struct {
        StartedAt    time.Time `json:"started_at"`
        DurationDays int       `json:"duration_days"` // 0=immediate, -1=indefinite, N=N days
    }

    // IsBaselineActive returns true if the baseline period is still active at now.
    func IsBaselineActive(state BaselineState, now time.Time) bool {
        if state.DurationDays == 0 {
            return false // immediate enforcement
        }
        if state.DurationDays < 0 {
            return true // indefinite audit-only
        }
        deadline := state.StartedAt.Add(time.Duration(state.DurationDays) * 24 * time.Hour)
        return now.Before(deadline)
    }

    SENTRY-001 logic (EventFileAccess):
    - If !isEditorDescendant(event.PID, tree): skip
    - If !isSensitivePath(event.FilePath): skip
    - Append RuleWindowEntry{PID, FilePath, now} to state.CredAccessByPID[event.PID]

    - If !isEditorDescendant(event.PID, tree): skip

    SENTRY-003 logic (EventNetworkConnect):
    - If !isEditorDescendant(event.PID, tree): skip
    - Record in state.PhoneHomeByPID; expire entries older than PhoneHomeWindowMin
    - Fire SENTRY-003 alert on first qualifying connection (one connection = one alert per window)

    SENTRY-004 logic (post-alert correlation):
    - After any SENTRY-001/002/003 alert fires: check inventory.RecentExtensions for any entry with install time within FreshExtWindowMin of now
    - If found: fire additional SENTRY-004 alert; set CorrelatedExtension to the extension ID

    SENTRY-005 logic (fusion):
    - On EventNetworkConnect for an editor-descended PID: check state.CredAccessByPID[event.PID] for any entry within ExfilFusionWindowMin of now AND check inventory for extension installed within ExfilFusionWindowMin

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| eBPF/fanotify events → correlation engine | Event data arrives from kernel; must not cause unbounded allocation or panic on crafted events |
| Baseline JSON file → BaselineState | File could be tampered; malformed JSON must not crash daemon |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-02-01 | Denial of Service | Unbounded RuleState growth if entries never expire | mitigate | expireWindow called on every EvaluateEvent invocation; window entries for PIDs not seen in > 10 min are discarded |
| T-05-02-02 | Tampering | Attacker writing a zero-duration baseline file to suppress alerts | mitigate | LoadBaseline missing file returns 7-day default; DurationDays=0 means immediate enforcement (most restrictive); tampering the file to DurationDays=0 makes rules stricter, not weaker |
| T-05-02-03 | Tampering | Rule SENTRY-004 falsely correlating unrelated extension with alert | accept | SENTRY-004 is a correlation hint, not a block; it produces an alert record only; false positives in audit-only mode are acceptable for v1 |
| T-05-02-04 | Elevation of Privilege | isEditorDescendant walking infinite PPID loops in crafted tree | mitigate | Loop broken on missing tree entry or pid==0; max 8-ancestor ParentChain cap prevents unbounded iteration |
</threat_model>
