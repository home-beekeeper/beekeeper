    Implement `func correlationEngineLoop(ctx context.Context, events <-chan sentry.SentryEvent, state *daemonState, auditWriter *audit.Writer, baselinePath string)`:
    - baseline, _ := sentry.LoadBaseline(baselinePath); if file did not exist, baseline now defaults to 7-day starting now and we need to save it once: sentry.SaveBaseline(baselinePath, baseline) (ignore error).
    - ruleState := sentry.NewRuleState().
    - tree := make(map[uint32]sentry.ProcessNode).
    - Loop:
      ```
      for {
          select {
          case <-ctx.Done():
              return
          case ev, ok := <-events:
              if !ok { return }
              atomic.AddUint64(&state.eventsProcessed, 1)
              if ev.Kind == sentry.EventProcessCreate {
                  tree[ev.PID] = sentry.ProcessNode{
                      PID: ev.PID, PPID: ev.PPID, UID: ev.UID,
                      Exe: ev.Exe, Cmdline: ev.Cmdline, SeenAt: ev.WallTime,
                  }
                  // GC: drop entries older than 10 min
                  cutoff := time.Now().Add(-10 * time.Minute)
                  for pid, node := range tree {
                      if node.SeenAt.Before(cutoff) { delete(tree, pid) }
                  }
              }
              alerts := sentry.EvaluateEvent(ev, ruleState, tree, sentry.InventorySnapshot{}, sentry.RuleConfig{}, baseline, time.Now().UTC())
              for _, alert := range alerts {
                  rec := alertToAuditRecord(alert)
                  if err := auditWriter.Write(rec); err != nil {
                      fmt.Fprintf(os.Stderr, "beekeeper sentry: audit write: %v\n", err)
                  }
              }
          }
      }
      ```
