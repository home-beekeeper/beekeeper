```go
// 4-byte big-endian uint32 length prefix + JSON payload
func Encode(w io.Writer, v any) error {
    data, err := json.Marshal(v)
    // ... if len(data) > maxMessageSize return ErrMessageTooLarge
    var hdr [4]byte
    binary.BigEndian.PutUint32(hdr[:], uint32(len(data)))
    w.Write(hdr[:])
    w.Write(data)
}

func Decode(r io.Reader, v any) error {
    var hdr [4]byte
    io.ReadFull(r, hdr[:])
    size := binary.BigEndian.Uint32(hdr[:])
    // ... if size > maxMessageSize return ErrMessageTooLarge
    payload := make([]byte, size)
    io.ReadFull(r, payload)
    return json.Unmarshal(payload, v)
}
```

```go
type ScanKind string
const (
    ScanPrompt    ScanKind = "scan_prompt"
    ScanCode      ScanKind = "scan_code"
    ScanAlignment ScanKind = "scan_alignment"
)

type ScanRequest struct {
    Kind      ScanKind `json:"kind"`
    Content   string   `json:"content"`
    Context   string   `json:"context,omitempty"`
    RequestID string   `json:"request_id"`
}

type ScanResult string
const (
    ResultClean     ScanResult = "clean"
    ResultInjection ScanResult = "injection"
    ResultUnsafe    ScanResult = "unsafe"
    ResultHijacked  ScanResult = "hijacked"
)

type ScanResponse struct {
    RequestID  string     `json:"request_id"`
    Result     ScanResult `json:"result"`
    Confidence float64    `json:"confidence"`
    Reason     string     `json:"reason,omitempty"`
    LatencyMS  int64      `json:"latency_ms"`
    Error      string     `json:"error,omitempty"`
}

var ErrMessageTooLarge = errors.New("llamafirewall: message exceeds 1MB limit")
var ErrNotSupported    = errors.New("llamafirewall: not supported on this platform")
const maxMessageSize   = 1024 * 1024  // 1MB
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-06-02-01 | Denial of Service | Decode allocating 1MB+ for malicious length prefix | mitigate | 1MB hard cap in Decode; ErrMessageTooLarge before make([]byte, size); fuzz test covers 0xFFFFFFFF prefix |
| T-06-02-02 | Tampering | Malformed JSON in ScanResponse causing json.Unmarshal panic | mitigate | json.Unmarshal returns error, never panics; Decode propagates error; caller checks error before using response |
