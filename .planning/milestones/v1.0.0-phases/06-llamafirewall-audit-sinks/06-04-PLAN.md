    ```go
    type LatencyTracker struct {
        mu     sync.Mutex
        count  int64
        sumMS  int64
        p95buf [100]int64  // ring buffer for last 100 latencies
        head   int         // next write position
        filled bool        // true once all 100 slots written at least once
    }
    func (t *LatencyTracker) Record(ms int64)
    func (t *LatencyTracker) P95() int64
    func (t *LatencyTracker) Mean() float64
    ```

    Record: lock mu; increment count and sumMS; write ms into p95buf[head % 100]; increment head; if head >= 100, set filled=true.

    P95: lock mu; if count == 0 return 0; copy relevant slice from p95buf (all 100 if filled, else head elements); sort a copy; return element at index int(float64(len)*0.95) using math.Floor. Do NOT mutate p95buf during P95 computation.

    Mean: lock mu; if count == 0 return 0; return float64(sumMS) / float64(count).

    ```go
    type Client struct {
        conn    net.Conn
        timeout time.Duration
    }

    func Dial(sockPath string, timeout time.Duration) (*Client, error) {
        conn, err := net.DialTimeout("unix", sockPath, timeout)
        if err != nil { return nil, err }
        return &Client{conn: conn, timeout: timeout}, nil
    }

    func (c *Client) Scan(req ScanRequest) (ScanResponse, error) {
        c.conn.SetWriteDeadline(time.Now().Add(c.timeout))
        if err := Encode(c.conn, req); err != nil {
            return ScanResponse{}, err
        }
        c.conn.SetReadDeadline(time.Now().Add(c.timeout))
        var resp ScanResponse
        if err := Decode(c.conn, &resp); err != nil {
            return ScanResponse{}, err
        }
        return resp, nil
    }

    func (c *Client) Close() error { return c.conn.Close() }
    ```

    ```go
    const windowsPipePath = `\\.\pipe\beekeeper-llamafirewall`

    type Client struct {
        handle  windows.Handle
        timeout time.Duration
    }

    func Dial(sockPath string, timeout time.Duration) (*Client, error) {
        // sockPath is ignored on Windows; always use windowsPipePath
        path, _ := windows.UTF16PtrFromString(windowsPipePath)
        h, err := windows.CreateFile(
            path,
            windows.GENERIC_READ|windows.GENERIC_WRITE,
            0, nil,
            windows.OPEN_EXISTING,
            windows.FILE_ATTRIBUTE_NORMAL,
            0,
        )
        if err != nil {
            return nil, fmt.Errorf("open named pipe %s: %w", windowsPipePath, err)
        }
        return &Client{handle: h, timeout: timeout}, nil
    }

    func (c *Client) Scan(req ScanRequest) (ScanResponse, error) {
        // Encode req to JSON, write via windows.WriteFile; read response via windows.ReadFile
        // Use Encode/Decode from proto.go by wrapping the handle in a custom io.ReadWriter
        rw := &pipeReadWriter{handle: c.handle}
        if err := Encode(rw, req); err != nil { return ScanResponse{}, err }
        var resp ScanResponse
        if err := Decode(rw, &resp); err != nil { return ScanResponse{}, err }
        return resp, nil
    }

    func (c *Client) Close() error {
        return windows.CloseHandle(c.handle)
    }

    type pipeReadWriter struct{ handle windows.Handle }

    func (p *pipeReadWriter) Write(b []byte) (int, error) {
        var n uint32
        err := windows.WriteFile(p.handle, b, &n, nil)
        return int(n), err
    }

    func (p *pipeReadWriter) Read(b []byte) (int, error) {
        var n uint32
        err := windows.ReadFile(p.handle, b, &n, nil)
        return int(n), err
    }
    ```

    ```python
    #!/usr/bin/env python3
    """LlamaFirewall sidecar — supervised by beekeeper."""
    import json, os, socket, struct, sys

    SOCK_PATH = os.path.expanduser("~/.beekeeper/llamafirewall.sock")
    MAX_MSG = 1024 * 1024  # 1MB

    def recv_msg(conn):
        hdr = conn.recv(4, socket.MSG_WAITALL)
        if len(hdr) < 4: return None
        size = struct.unpack(">I", hdr)[0]
        if size > MAX_MSG: raise ValueError(f"message too large: {size}")
        data = conn.recv(size, socket.MSG_WAITALL)
        return json.loads(data)

    def send_msg(conn, obj):
        data = json.dumps(obj).encode()
        conn.sendall(struct.pack(">I", len(data)) + data)

    def handle_scan(req: dict) -> dict:
        kind = req.get("kind", "")
        content = req.get("content", "")
        request_id = req.get("request_id", "")
        import time
        start = time.time()
        # Load models lazily on first call
        result, confidence, reason = "clean", 0.0, ""
        try:
            if kind == "scan_prompt":
                from llamafirewall import LlamaFirewall, UserMessage, ScanDecision
                lf = LlamaFirewall()
                decision = lf.scan(UserMessage(role="user", content=content))
                if decision.decision != ScanDecision.ALLOW:
                    result, confidence, reason = "injection", float(decision.score or 0.9), str(decision.reason or "")
            elif kind == "scan_code":
                # CodeShield placeholder — replace with real CodeShield call
                result = "clean"
            elif kind == "scan_alignment":
                result = "clean"
        except Exception as e:
            return {"request_id": request_id, "result": "clean", "confidence": 0.0,
                    "latency_ms": 0, "error": str(e)}
        latency_ms = int((time.time() - start) * 1000)
        return {"request_id": request_id, "result": result, "confidence": confidence,
                "reason": reason, "latency_ms": latency_ms}

    def main():
        if os.path.exists(SOCK_PATH): os.unlink(SOCK_PATH)
        srv = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
        srv.bind(SOCK_PATH); os.chmod(SOCK_PATH, 0o600)
        srv.listen(1)
        print(f"llamafirewall_sidecar: listening on {SOCK_PATH}", flush=True)
        while True:
            conn, _ = srv.accept()
            try:
                req = recv_msg(conn)
                if req is None: continue
                resp = handle_scan(req)
                send_msg(conn, resp)
            except Exception as e:
                print(f"llamafirewall_sidecar: error: {e}", file=sys.stderr, flush=True)
            finally:
                conn.close()

    if __name__ == "__main__":
        main()
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Go supervisor → Python sidecar process | Python process is supervised but untrusted; sidecar crash must trigger fail-closed, not silent allow |
| Python sidecar → LlamaFirewall models | Model inference result is trusted only as a signal; the Go side makes the final block/allow decision |
| Unix socket (llamafirewall.sock) → Go Client | Socket created with 0600 perms by sidecar; only the beekeeper user can connect |
| CLI llamafirewall enable/disable → config.json | User-level config change; no privilege escalation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-04-01 | Elevation of Privilege | Python sidecar running as a subprocess with inherited env | mitigate | Sidecar runs as the same user as beekeeper (not root); no setuid; supervisor does not grant additional capabilities |
| T-06-04-02 | Denial of Service | Supervisor restart loop spinning after MaxRetries | mitigate | After MaxRetries the goroutine exits and degraded=true; no further restart attempts; Scan returns error immediately |
| T-06-04-03 | Tampering | Malicious scan response from compromised Python sidecar | mitigate | ScanResponse is parsed via Decode (1MB cap, JSON unmarshal); the Go side applies the final block/warn decision; a false "clean" from the sidecar only affects LLMF signal, not the primary policy engine |
| T-06-04-04 | Information Disclosure | ScanRequest Content containing source code sent to Python process | accept | Python process is local, supervised, same-user; content is already visible to the agent; sidecar loads models locally, no network calls |
| T-06-04-05 | Denial of Service | Sample rate = 0 allows all content unscannable | accept | SampleRate is a documented opt-out configured by the developer; default 1.0 (scan all) |
| T-06-04-06 | Spoofing | beekeeper llamafirewall status reading stale PID from state.json | accept | Status command validates process liveness with os.Signal(0); stale PID shows "Not running" |
