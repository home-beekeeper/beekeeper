    func (m *MultiSink) Write(rec AuditRecord) error {

            }
        }
        return lastErr

    func (m *MultiSink) Close() error {
        var lastErr error

            if err := s.Close(); err != nil {

            }
        }
        return lastErr
    }

    ```go
    type OTLPSink struct {
        endpoint string
        client   *http.Client
        mu       sync.Mutex
        batch    []AuditRecord
    }

    func NewOTLPSink(endpoint string) *OTLPSink {
        return &OTLPSink{
            endpoint: endpoint,
            client:   &http.Client{Timeout: 10 * time.Second},
            batch:    make([]AuditRecord, 0, 100),
        }
    }

    func (s *OTLPSink) Write(rec AuditRecord) error {
        s.mu.Lock()
        defer s.mu.Unlock()
        s.batch = append(s.batch, rec)
        if len(s.batch) >= 100 {
            return s.flushLocked()
        }
        return nil
    }

    func (s *OTLPSink) Close() error {
        s.mu.Lock()
        defer s.mu.Unlock()
        if len(s.batch) > 0 {
            return s.flushLocked()
        }
        return nil
    }

    func (s *OTLPSink) flushLocked() error { ... }
    ```

    type HTTPSink struct {
        endpoint string
        client   *http.Client
    }

    func NewHTTPSink(endpoint string) *HTTPSink {
        return &HTTPSink{
            endpoint: endpoint,
            client:   &http.Client{Timeout: 5 * time.Second},

    func (s *HTTPSink) Write(rec AuditRecord) error {
        data, _ := json.Marshal(rec)
        data = append(data, '\n')
        req, err := http.NewRequest(http.MethodPost, s.endpoint, bytes.NewReader(data))
        if err != nil {
            fmt.Fprintf(os.Stderr, "beekeeper https sink: request error: %v\n", err)
            return nil // fire-and-forget

        if err != nil {
            fmt.Fprintf(os.Stderr, "beekeeper https sink: post error: %v\n", err)
            return nil // fire-and-forget
        }
        resp.Body.Close()
        return nil
    }

    }

    func NewSyslogSink(address string) (*SyslogSink, error) {
        // Parse address: "proto:host:port" or "host:port" (default udp)
        // ...
        w, err := syslog.Dial(network, addr, syslog.LOG_LOCAL0|syslog.LOG_INFO, "beekeeper")
        if err != nil {
            return nil, fmt.Errorf("syslog dial %q: %w", address, err)
        }
        return &SyslogSink{writer: w}, nil

    func (s *SyslogSink) Write(rec AuditRecord) error {
        // RFC 5424: <134>1 TIMESTAMP HOSTNAME APP-NAME PROCID MSGID [beekeeper@0 decision="..." tool="..." agent="..."] MSG
        hostname, _ := os.Hostname()

        if ts == "" { ts = time.Now().UTC().Format(time.RFC3339) }

        return err
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Go process → syslog UDP/TCP | Audit data leaves machine; must be explicit opt-in with documented warning |
| Go process → OTLP HTTP endpoint | Audit data leaves machine; fire-and-forget errors do not affect local write |
| Go process → HTTPS POST endpoint | Audit data leaves machine; same fire-and-forget semantics |
| Config file → AuditConfig sinks field | User-controlled list; unknown sink names are silently ignored (no attack surface from unknown values) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03-01 | Information Disclosure | Remote audit sinks sending data off-machine without user awareness | mitigate | All remote sinks off by default; NewMultiSink prints one-time stderr warning when any non-file sink is created |
| T-06-03-02 | Denial of Service | OTLPSink batch growing unbounded if flush endpoint is unreachable | mitigate | Batch is reset after every flush attempt (even on error); maximum 100 records before forced flush; Write returns nil on error (does not block caller) |
| T-06-03-03 | Denial of Service | HTTPSink blocking indefinitely on slow HTTPS endpoint | mitigate | 5s per-request timeout on http.Client; fire-and-forget on error |
| T-06-03-04 | Tampering | Syslog address field accepting arbitrary proto:host:port | mitigate | Only "udp" and "tcp" are accepted as network protocol prefix; syslog.Dial validates the address format |
| T-06-03-05 | Information Disclosure | AuditRecord in OTLP body.stringValue containing credentials not redacted at sink time | accept | Redaction is applied before records are written to the Writer (check/handler.go); sinks receive already-redacted records |
</threat_model>
