    ```go
    // ShouldScanPrompt returns true for tool names whose outputs flow into agent
    // context and should be scanned by PromptGuard 2 (LLMF-02).
    func ShouldScanPrompt(toolName string) bool {
        switch toolName {
        case "web_search", "read_file":
            return true
        }
        return false
    }

    // ShouldScanCode returns true for tool names that write code and should be
    // evaluated by CodeShield (LLMF-03).
    func ShouldScanCode(toolName string) bool {
        return toolName == "write_file"
    }

    // BuildWarningPayload returns the JSON warning payload injected into agent
    // context when a scan detects injection, unsafe code, or alignment hijacking.
    // The payload follows the structure from 06-CONTEXT.md §LLMF-02.
    func BuildWarningPayload(resp ScanResponse) []byte {
        alertType := "prompt_injection"
        reason := resp.Reason
        switch resp.Result {
        case ResultUnsafe:
            alertType = "code_unsafe"
            if reason == "" { reason = "CodeShield detected insecure code pattern" }
        case ResultHijacked:
            alertType = "alignment_hijack"
            if reason == "" { reason = "AlignmentCheck detected goal hijacking signal" }
        default:
            if reason == "" { reason = "PromptGuard 2 detected indirect prompt injection attempt in tool output" }
        }
        payload := map[string]any{
            "beekeeper_alert":           true,
            "alert_type":               alertType,
            "confidence":               resp.Confidence,
            "reason":                   reason,
            "original_content_redacted": true,
        }
        data, _ := json.Marshal(payload)
        return data
    }
    ```

    ```go
    // ScanProxiedResponse runs PromptGuard 2 on a proxied MCP tool response body.
    // Returns the (possibly replaced) body and whether injection was detected.
    // If sup is nil or the tool is not scan-eligible, returns body unchanged.
    func ScanProxiedResponse(ctx context.Context, toolName string, body []byte, sup *llamafirewall.Supervisor) ([]byte, bool) {
        if sup == nil || !llamafirewall.ShouldScanPrompt(toolName) {
            return body, false
        }
        resp, err := sup.Scan(ctx, llamafirewall.ScanRequest{
            Kind:      llamafirewall.ScanPrompt,
            Content:   string(body),
            Context:   toolName,
            RequestID: fmt.Sprintf("gw-%d", time.Now().UnixNano()),
        })
        if err != nil || resp.Result == llamafirewall.ResultClean {
            return body, false
        }
        // Replace body with structured warning
        warning := llamafirewall.BuildWarningPayload(resp)
        return warning, true
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Tool output (web_search/read_file) → PromptGuard 2 scan | Tool output is untrusted external content; injection detection gates what enters agent context |
| Agent code write (write_file) → CodeShield scan | Agent-generated code is potentially malicious; CodeShield result informs block/warn decision |
| ScanResponse from sidecar → handler decision | Sidecar response is treated as an advisory signal; fail-closed applies when sidecar unavailable |
| Gateway proxied response body → ScanProxiedResponse | Upstream MCP server response is untrusted; PromptGuard 2 runs before result is forwarded to agent |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-05-01 | Tampering | Prompt injection via web_search result bypassing LLMF | mitigate | ShouldScanPrompt("web_search") returns true; sidecar unavailability with fail_closed blocks the result; handler replaces injection with structured warning |
| T-06-05-02 | Tampering | Indirect injection via read_file content replacing agent context | mitigate | ShouldScanPrompt("read_file") returns true; same mitigation as T-06-05-01 |
| T-06-05-03 | Elevation of Privilege | CodeShield false negative allowing malicious code write | accept | CodeShield is a best-effort signal; the primary defense is the policy engine (Plan 01–02); CodeShield adds defense-in-depth. Documented limitation. |
| T-06-05-04 | Repudiation | LLMF scan result not recorded in audit log | mitigate | LLMFScanned, LLMFResult, LLMFConfidence, LLMFLatencyMS fields in AuditRecord; llmf_alert records written on injection/unsafe hits |
| T-06-05-05 | Denial of Service | LLMF scan adding latency to every tool call on the PostToolUse path | accept | PostToolUse hooks must not block the agent; LLMF scan runs on PostToolUse (result available), not PreToolUse (pre-execution); P95 target is 100ms per PromptGuard 2 benchmarks |
| T-06-05-06 | Information Disclosure | Original injected content logged to audit before redaction | mitigate | RedactRecord applied before audit write; LLMF fields do not store the original content — only the scan result and confidence |
</threat_model>
