resolveCatalogHealthy body (verbatim from 06-PATTERNS.md § resolveCatalogHealthy helper):
  func resolveCatalogHealthy(cacheDir string) bool {
      if cacheDir == "" { return true }
      statePath := filepath.Join(filepath.Dir(cacheDir), "state.json")
      state, err := catalog.LoadState(statePath)
      if err != nil { return true }
      if src, ok := state.Sources["bumblebee"]; ok { return !src.Degraded }
      return true
  }
