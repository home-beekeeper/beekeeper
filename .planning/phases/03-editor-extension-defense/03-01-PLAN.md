```go
// extract pulls (ecosystem, package, version) from a tool call's input.
func extract(input map[string]any) (ecosystem, pkg, version string, ok bool) {
    if input == nil { return "", "", "", false }
    if eco, pkgRaw, ok2 := directPackage(input); ok2 {
        ver, _ := input["version"].(string)
        return eco, normalize(pkgRaw), strings.TrimSpace(ver), true
    }
    if cmd, ok2 := input["command"].(string); ok2 {
        return extractFromCommand(cmd)   // <-- editor check inserted BEFORE this
    }
    return "", "", "", false
}

// existing helpers available for reuse (DO NOT reimplement):
func firstPackageToken(rest string) string  // first non-flag whitespace token
func splitVersion(token string) (name, version string)  // splits on LAST "@"
func normalize(pkg string) string  // strings.ToLower(strings.TrimSpace(pkg))
```

The ecosystem string "editor-extension" already flows correctly through the rest of the
engine: the selftest corpus entry `nrwl.angular-console` uses Ecosystem "editor-extension"
and matches via the existing index path. No engine changes beyond extract() are needed.
